Quick Start

# Clone and build
git clone https://github.com/Dr-H-PhD/h-lang.git
cd h-lang
go build -o hlc ./cmd/hlc

# Compile and run
./hlc -run examples/hello.hl

H-lang Syntax Preview

# hello.hl
function main() {
    x := 42;
    const PI := 3.14159;

    # Arrays and for-range
    nums := [5]int{1, 2, 3, 4, 5};
    for i, v := range nums {
        print(v);
    }

    # Heap allocation with defer cleanup
    user := alloc(User);
    defer free(user);  # Runs at function exit

    user.name = "Achraf";
    print(user.greet());

    # Classic for loop
    for i := 0; i < 10; i++ {
        print(i);
    }
}
